EDU01A_longer <-
  EDU01A_long |>
  mutate(two_digit_year= (substr(EDU_combined, start=8, stop = 9)) ,
  year_dbl = as.double(two_digit_year), 
  year= if_else(year_dbl >25, 1900 + year_dbl, year_dbl <=25 + 2000),
